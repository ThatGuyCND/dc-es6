// Ternary in template literal
const isPizza = true;
const cheer = () => `
  YEAAAAAA!
`;

const template = () => `
  <div>
    ${isPizza ? `${cheer()}` : `I'm so sad. :(`}
  </div>
`;

const nextLink = document.querySelector(`.next`);
nextLink.insertAdjacentHTML(`afterend`, template());
